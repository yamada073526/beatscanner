### vision-eval 3 ticker × 3 run mean (PDCA measure)
```bash
cd frontend
for ticker in AAPL MSFT NVDA; do
  for run in 1 2 3; do
    ANTHROPIC_API_KEY=$(grep ANTHROPIC_API_KEY ../backend/.env | cut -d= -f2 | tr -d '"') \
      node scripts/snap-vision-eval.mjs --ticker $ticker \
      --out .visual/eval-${ticker}-<phase>-run${run}.json
  done
done

# 集計
python3 -c "
import json, glob
def mean(phase):
    files = sorted(glob.glob(f'.visual/eval-*-{phase}-run*.json'))
    sc = {}
    for f in files:
        s = json.load(open(f)).get('scores', {})
        for k in ['typography','spacing','color','motion','aman','overall']:
            sc.setdefault(k, []).append(s.get(k, 0))
    return {k: sum(v)/len(v) for k,v in sc.items()}

baseline = mean('v97end')  # or 'J2lite' を最新 baseline として使う
cur = mean('<phase>')
for k in baseline:
    d = cur[k] - baseline[k]
    print(f'{k}: {baseline[k]:.1f} → {cur[k]:.1f} (Δ {d:+.2f})')
"
```
